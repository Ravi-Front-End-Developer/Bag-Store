DBMS DESIGN

-- 1. User Table
CREATE TABLE users (
    other fields are already created
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Personal Info Binding
CREATE TABLE profiles (
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    full_name VARCHAR(100),
    phone VARCHAR(20),
    profile_pic_url TEXT,
    bio TEXT,
    PRIMARY KEY (user_id)
);

-- 3. Payment Info Binding
CREATE TABLE payment_methods (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    card_type VARCHAR(20), -- e.g., 'Visa'
    last_four VARCHAR(4),
    stripe_customer_id TEXT, -- Binding to external payment gateway
    is_default BOOLEAN DEFAULT false
);

-- 4. Order Binding
CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(id),
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(50), -- e.g., 'Delivered', 'Pending'
    total_price DECIMAL(10, 2)
);
